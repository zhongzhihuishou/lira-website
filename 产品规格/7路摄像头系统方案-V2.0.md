### 多路视频流处理框架

```python
import cv2
import threading
from queue import Queue

class MultiCameraSystem:
    def __init__(self):
        # 摄像头配置
        self.camera_configs = {
            'cam01_global': {'id': 0, 'fps': 120, 'res': (1920, 1080)},
            'cam02_recognize': {'id': 1, 'fps': 60, 'res': (1920, 1080)},
            'cam03_recognize': {'id': 2, 'fps': 60, 'res': (1920, 1080)},
            'cam04_peopleflow': {'id': 3, 'fps': 30, 'res': (3840, 2160)},
            'cam05_face': {'id': 4, 'fps': 60, 'res': (1280, 720)},
            'cam06_monitor': {'id': 5, 'fps': 30, 'res': (1920, 1080)},
            'cam07_monitor': {'id': 6, 'fps': 30, 'res': (1920, 1080)},
        }
        
        # 帧队列
        self.frame_queues = {name: Queue(maxsize=30) 
                            for name in self.camera_configs}
        
        # 初始化摄像头
        self.captures = {}
        self._init_cameras()
        
        # 启动处理线程
        self._start_processing_threads()
    
    def _init_cameras(self):
        """初始化所有摄像头"""
        for name, config in self.camera_configs.items():
            cap = cv2.VideoCapture(config['id'])
            cap.set(cv2.CAP_PROP_FPS, config['fps'])
            cap.set(cv2.CAP_PROP_FRAME_WIDTH, config['res'][0])
            cap.set(cv2.CAP_PROP_FRAME_HEIGHT, config['res'][1])
            self.captures[name] = cap
    
    def _capture_worker(self, name):
        """摄像头采集线程"""
        cap = self.captures[name]
        queue = self.frame_queues[name]
        
        while True:
            ret, frame = cap.read()
            if ret:
                if queue.full():
                    queue.get()  # 丢弃旧帧
                queue.put(frame)
    
    def _start_processing_threads(self):
        """启动所有采集线程"""
        for name in self.camera_configs:
            t = threading.Thread(target=self._capture_worker, args=(name,))
            t.daemon = True
            t.start()
```
